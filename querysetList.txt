#Be sure to import a model first before making queries
from .models import ModelName

#all() - Retrieve all objects from table
queryset = ModelName.objects.all()

#get(attribuet='value') - Retrieve a single object based on a matched attribute
queryItem=ModelName.objects.get(attribute="value")

#filter(attribute='value) - Returns all items from table that match a particular attribute value
filteredQueryset = ModelName.objects.filter(attribute='value')
                                    .filter(attribue__startswith ='value')
                                    .filter(attribue__contains ='value')
                                    .filter(attribue__icontains ='value')
                                    .filter(attribue__gt ='value')
                                    .filter(attribue__gte ='value')
                                    .filter(attribue__lt ='value')
                                    .filter(attribue__lte ='value')

#exclude(attribute='value') - Excludes any object matching a particular filter
excludedQuerySet = ModelName.objects.exclude(attribute='value')

#order_by() - Order a queryset by a particular attribute. Multiple parameters are allowed
orderedQuerySet = Project.object.filter(title = 'first project').objects.order_by('attribute', 'otherAttribute')

#order can be reversed by adding "-" before the attribute name
reversedOrderQuerySet =Project.object.filter(title = 'first project').objects.order_by('-attribute', '-otherAttribute')

newObject = ModelName.objects.create(**kwargs)
